\subsection{Compact-by-compact transfer (T5)}\label{sec:T5-compact}

\subsection*{Standing analytic inputs (uniform version)}
For each $K>0$ we use the uniform analytic data from Sections~\ref{sec:A3} and~\ref{sec:rkhs-prime-cap}:
\begin{itemize}
  \item[\textbf{(A3)}] Uniform Archimedean floor: $\min_{\theta\in\TT} P_A(\theta) \ge c_* \ge 811/1000$
  (Lemma~\ref{lem:uniform-arch-floor}).
  \item[\textbf{(A3.b)}] Discretization control: for all $M\ge M_0^{\mathrm{unif}}$,
  \[
    C_{\mathrm{SB}}\,\omega_{P_A}\!\Big(\frac{\pi}{M}\Big)\ \le\ \frac{c_*}{2}
  \]
  (Corollary~\ref{cor:uniform-discretisation}).
  \item[\textbf{(RKHS)}] Uniform prime contraction: at $t_0=7/10$,
  \[
    \|{\mathcal T}_P\|\ \le\ \rho(t_0)\ \le\ \frac{1971}{50000}\ <\ \frac{c_*}{4}
  \]
  (Theorem~\ref{thm:rkhs-tstar} and Corollary~\ref{cor:pcu-uniform}).
\end{itemize}

We also recall the density/continuity interface on $W_K$:
\begin{itemize}
  \item[\textbf{(A1$'$)}] The Fej\'er$\times$heat cone is dense in $W_K$.
  \item[\textbf{(A2)}] $Q$ is continuous on $W_K$; specifically $|Q(\Phi)-Q(\Psi)|\le L_Q(K)\|\Phi-\Psi\|_\infty$.
\end{itemize}

\begin{lemma}[Grid-lift inequality]\label{lem:T5p-grid}
For every $K>0$ and $M\ge M_0^{\mathrm{unif}}$,
\[
  \lambda_{\min}\!\big(T_M[P_A]-{\mathcal T}_P\big)
  \ \ge\ c_*\ -\ C_{\mathrm{SB}}\,\omega_{P_A}\!\Big(\tfrac{\pi}{M}\Big)\ -\ \|{\mathcal T}_P\|
  \ \ge\ \frac{c_*}{4}.
\]
\end{lemma}

\begin{proof}
Combine the uniform Archimedean floor $\min P_A \ge c_*$ with:
\begin{itemize}
  \item Discretization: $C_{\mathrm{SB}}\,\omega_{P_A}(\pi/M) \le c_*/2$ for $M\ge M_0^{\mathrm{unif}}$.
  \item Prime cap: $\|{\mathcal T}_P\| \le \rho(t_0) < c_*/4$.
\end{itemize}
Then $\lambda_{\min} \ge c_* - c_*/2 - c_*/4 = c_*/4$.
\end{proof}

\begin{theorem}[T5: uniform compact transfer]\label{thm:T5-compact}
For every $K>0$ one has
\[
  \lambda_{\min}\!\big(T_{M}[P_A]-{\mathcal T}_P\big)\ \ge\ \frac{c_*}{4}\ \ge\ \frac{811}{4000}\ >\ 0.2
\]
for all $M\ge M_0^{\mathrm{unif}}$.
In particular, $Q(\Phi)\ge0$ on $W_K$ for all $K>0$. Hence $Q\ge0$ on $\bigcup_{K>0} W_K$, i.e., on the full Weil class.
\end{theorem}

\begin{proof}
By Lemma~\ref{lem:T5p-grid}, the finite Toeplitz form $T_M[P_A]-{\mathcal T}_P$ has positive
minimum eigenvalue $\ge c_*/4 > 0$ for all $M\ge M_0^{\mathrm{unif}}$.
Positivity on the Fej\'er$\times$heat cone follows.
Then (A1$'$)--(A2) extend $Q\ge0$ from the dense cone to all of $W_K$.
Taking the union over $K$ gives the claim.
\end{proof}

\begin{remark}[No K-dependent schedules]
The uniform approach eliminates the need for K-dependent schedules $c_0(K)$, $t^\star(K)$, $M^\star(K)$.
The same uniform parameters $c_*\ge 811/1000$, $t_0=7/10$, and $M_0^{\mathrm{unif}}$
work for all compacts $[-K,K]$ simultaneously.
\end{remark}
