% Theorem A1' --- Local Cone Density on Frequency Compacts

We work on $C^+_{\mathrm{even}}([-K,K])$ with the uniform norm $\|\cdot\|_\infty$.
Convolution with the Fejér kernel and subsequent heat smoothing preserve evenness and nonnegativity.

\begin{theorem}[A1' --- density]\label{thm:A1-density}
For every compact $[-K,K]$ the cone $\{\text{Fejér}*\text{heat approximants}\}$ is dense in $C^+_{\mathrm{even}}([-K,K])$ in $\|\cdot\|_\infty$.
\end{theorem}
\begin{proof}
Fejér kernels form a positive approximation identity on $\mathbb T$; heat flow preserves positivity and evenness, hence the uniform limit remains in the cone.
\end{proof}

\begin{remark}[PW reinforcement]
On $[-K,K]$ the heat kernel satisfies $\widehat{\rho_t}(s)=e^{-4\pi^2 t s^2}\ge e^{-4\pi^2 t K^2}>0$, hence the convolution with $\rho_t$ is invertible on the compact in the PW metric. Together with the Fej\'er (positive) hat interpolation and a Weierstrass/Fej\'er--Riesz approximation step in $\|\cdot\|_{\infty}$, this yields the cone density in $W_{\mathrm{PW},K}$ with explicit error control; the constants enter only via $e^{-4\pi^2 t K^2}$ and the mesh parameter in the hat partition of unity.
\end{remark}

\begin{lemma}[Compact support convolution reduction]\label{lem:convolution-compact-support}
Let $f\in C_c(\mathbb{R})$ be compactly supported with $\mathrm{supp}(f)\subseteq[-L,L]$, and let $g:\mathbb{R}\to\mathbb{R}$ be any function. Then for all $x\in\mathbb{R}$,
\[
(f*g)(x) = \int_{\mathbb{R}} f(y)\,g(x-y)\,dy = \int_{-L}^{L} f(y)\,g(x-y)\,dy.
\]
\end{lemma}
\begin{proof}
Since $f(y)=0$ for $|y|>L$, the integrand vanishes outside $[-L,L]$.
\end{proof}

\begin{theorem}[A1']\label{a1:thm:A1-local-density}
Let $K=[-R,R]$ with $R>0$. For $B>0$, $t>0$, $\tau\in[-R,R]$ define the even nonnegative frequency windows
\[
\Phi_{B,t,\tau}(\xi) := \Lambda_B(\xi-\tau)\, \rho_t(\xi-\tau) + \Lambda_B(\xi+\tau)\, \rho_t(\xi+\tau),
\]
where $\Lambda_B(x)=(1-|x|/B)_+$ and $\rho_t(x)=(4\pi t)^{-1/2} e^{-x^2/(4t)}$ (so $\int_{\mathbb R}\rho_t=1$, $\rho_t\ge0$). Let $\mathcal C$ be the closed convex cone generated by finite nonnegative combinations of $\{\Phi_{B,t,\tau}\}$ with $\tau\in[-R,R]$ and $B$ sufficiently large (depending on $R$). Then $\mathcal C$ is dense in $C^+_{\mathrm{even}}([-R,R])$ in the uniform norm.
\end{theorem}
\begin{proof}
Fix $f\in C^+_{\mathrm{even}}([-R,R])$ and $\varepsilon>0$. Extend $f$ by zero to a compactly supported $\widetilde f\in C_c(\mathbb R)$ with $\widetilde f=f$ on $[-R,R]$.

Step 1 (mollification). Since $\rho_t$ is a positive approximate identity, there exists $t\in(0,t_0]$ such that
\begin{equation}\label{eq:A1_local_density-formula-6}
\sup_{|\xi|\le R} |(\widetilde f*\rho_t)(\xi)-f(\xi)| < \varepsilon/3.
\end{equation}
Set $g:=\widetilde f*\rho_t$. Then $g\ge0$, $g\in C^{\infty}(\mathbb R)$ and $g$ is even.

Step 2 (positive Riemann sums). Choose a uniform partition $-R=\tau_0<\tau_1<\dots<\tau_N=R$ with mesh $\Delta$ small enough so that
\begin{equation}\label{eq:A1_local_density-formula-5}
g_R(\xi):=\sum_{j=0}^{N-1} g(\tau_j^*)\,\rho_t(\xi-\tau_j^*)\,(\tau_{j+1}-\tau_j)
\end{equation}
satisfies $\sup_{|\xi|\le R} |g_R(\xi)-g(\xi)|<\varepsilon/3$ for some choices $\tau_j^*\in[\tau_j,\tau_{j+1}]$. Because the coefficients $g(\tau_j^*)(\tau_{j+1}-\tau_j)$ are nonnegative, $g_R$ is a finite nonnegative combination of translates of $\rho_t$.

Step 3 (Fej\'er truncation). For any $|\xi|,|\tau|\le R$ one has $|\Lambda_B(\xi-\tau)-1|\le (|\xi|+|\tau|)/B\le 2R/B$. Choosing $B\ge B_0:=6R/\varepsilon$ ensures
\begin{equation}\label{eq:A1_local_density-formula-4}
\sup_{|\xi|\le R,\,|\tau|\le R} |\Lambda_B(\xi-\tau)-1| < \varepsilon/3.
\end{equation}
Define the symmetric Fej\'er$\times$heat mixture
\begin{equation}\label{eq:A1_local_density-formula-3}
h(\xi):=\sum_{j=0}^{N-1} g(\tau_j^*) (\tau_{j+1}-\tau_j)\,\Bigl[\Lambda_B(\xi-\tau_j^*)\,\rho_t(\xi-\tau_j^*)+\Lambda_B(\xi+\tau_j^*)\,\rho_t(\xi+\tau_j^*)\Bigr].
\end{equation}
Then for $|\xi|\le R$,
\begin{equation}\label{eq:A1_local_density-formula-2}
|h(\xi) - (g_R(\xi)+g_R(-\xi))| \le (\varepsilon/3) \sum_j g(\tau_j^*)(\tau_{j+1}-\tau_j) \le C_f\,\varepsilon/3,
\end{equation}
with $C_f=\int_{-R}^{R} g(\tau)\,d\tau$ finite. Rescaling $\varepsilon$ by $3\max(1,C_f)$ if necessary, we get
\begin{equation}\label{eq:A1_local_density-formula-1}
\sup_{|\xi|\le R} |h(\xi) - (g_R(\xi)+g_R(-\xi))| < \varepsilon/3.
\end{equation}

Step 4 (collect errors and evenness). Since $g$ is even, $g_R^{\mathrm{sym}}:=(g_R(\xi)+g_R(-\xi))/2$ is even and nonnegative, and by triangle inequality
\begin{equation}\label{eq:A1_local_density-formula}
\sup_{|\xi|\le R} |h(\xi)-f(\xi)| \le \sup|h-(g_R+g_R^{\phantom{1}})| + \sup|g_R-g| + \sup|g-f| < \varepsilon.
\end{equation}
By construction, $h$ is a finite nonnegative combination of $\Phi_{B,t,\tau_j^*}$ with $\tau_j^*\in[-R,R]$, hence $h\in\mathrm{cone}\{\Phi_{B,t,\tau}\}$. Taking closures in $\|\cdot\|_\infty$ yields density of $\mathcal C$ in $C^+_{\mathrm{even}}([-R,R])$.
\end{proof}

\begin{remark}[Parameter scaling in A1']\label{rem:a1-scaling}
The bandwidth $B$ and heat scale $t$ required in Theorem~\ref{a1:thm:A1-local-density}
depend on the compact $[-R,R]$ and the target accuracy $\varepsilon$.
Consequently, in the compact-by-compact transfer (T5) the schedules $K\mapsto B(K)$
and $K\mapsto t_{\mathrm{sym}}(K)$ are allowed to grow with $K$; no uniform bound
$\sup_K B(K)$ or $\sup_K t_{\mathrm{sym}}(K)$ is needed.
\end{remark}

\begin{lemma}[Fixed-$t_0$ cone density]\label{lem:a1-fixed-t-density}
Fix $K>0$ and $t_0>0$. Let $\mathcal C_K(t_0)$ be the uniform-closure on $[-K,K]$
of the conoid generated by Fej\'er$\times$heat atoms $\{\Phi_{B,t_0,\tau}:\ B>0,\ |\tau|+B\le K\}$
and their even symmetrizations. Then $\mathcal C_K(t_0)=C^+_{\mathrm{even}}([-K,K])$.
\label{a1:lem:A1-fix-t}
\end{lemma}
\begin{proof}
Let $f\in C^+_{\mathrm{even}}([-K,K])$ and $\varepsilon>0$. Choose a uniform grid with mesh $\delta>0$
and hats $H_j(\xi):=\Lambda_\delta(\xi-\tau_j)$. By positive piecewise-linear interpolation,
$h(\xi):=\sum_j f(\tau_j)H_j(\xi)$ satisfies $\|h-f\|_\infty<\varepsilon/3$ and $h\ge0$.
Since $\rho_{t_0}(\xi)$ is positive and Lipschitz on $[-K,K]$, for $\delta$ small one has
$\sup_{|u|\le\delta}\!|\rho_{t_0}(\xi+u)-\rho_{t_0}(\xi)|\le L_{t_0}\,\delta$. Set
$g(\xi):=\sum_j c_j\,\Phi_{\delta,t_0,\tau_j}(\xi)$ with $c_j:=f(\tau_j)/\rho_{t_0}(\tau_j)$.
Then $g(\xi)=\rho_{t_0}(\xi)\sum_j c_j\Lambda_\delta(\xi-\tau_j)+\mathcal O(L_{t_0}\,\delta)$,
hence $\|g-h\|_\infty\le(L_{t_0}\,\delta)\,\|c\|_{\ell^1}$; choosing $\delta$ small gives
$\|g-f\|_\infty<\varepsilon$. Evenization preserves nonnegativity. Taking the conoid-closure
yields the claim.
\end{proof}
