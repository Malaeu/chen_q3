The lemmas below provide the Frobenius drift control and Szeg\H{o}--B\"ottcher barrier
used in the uniform A3 bridge (Theorem~\ref{thm:A3}).

\begin{lemma}[Hoffman--Wielandt and Ky Fan guard]\label{lem:hw-kf}
Let $A,B\in\mathbb{C}^{M\times M}$ be Hermitian and set $E:=B-A$. Denote by $\lambda_i^{\downarrow}(A)$ the eigenvalues of~$A$ in non-increasing order. Then, for every $1\le k\le M$,
\[
  \sum_{i=1}^k \bigl|\lambda_i^{\downarrow}(B)-\lambda_i^{\downarrow}(A)\bigr|\ \le\ \sqrt{k}\,\|E\|_F,
\]
where $\|E\|_F=\sqrt{\mathrm{Tr}(E^*E)}$ is the Frobenius norm.
In particular
\[
  \bigl|\lambda_{\min}(B)-\lambda_{\min}(A)\bigr|\ \le\ \|E\|_F.
\]
\end{lemma}

\begin{proof}
The Hoffman--Wielandt inequality gives
\(\sum_i |\lambda_i(B)-\lambda_{\sigma(i)}(A)|^2\le \|E\|_F^2\) for a suitable permutation $\sigma$; see Horn--Johnson, \emph{Matrix Analysis} (2nd ed.), Thm.~7.4.9. Ky Fan majorisation (Cor.~7.3.5 loc.
cit.) implies
\(\sum_{i\le k} |\lambda_i^{\downarrow}(B)-\lambda_i^{\downarrow}(A)| \le \sum_{i\le k} \sigma_i(E)\), and Cauchy--Schwarz yields
\(\sum_{i\le k} \sigma_i(E) \le \sqrt{k}\,\|E\|_F\).
\end{proof}

\begin{corollary}[Frobenius slack for Toeplitz glue]\label{cor:toeplitz-frob}
Let $T_M[P]$ be a Toeplitz matrix and $\Delta T$ a perturbation with $\|\Delta T\|_F\le \varepsilon$. Then
\[
  \bigl|\lambda_{\min}(T_M[P+\Delta P]) - \lambda_{\min}(T_M[P])\bigr|\ \le\ \varepsilon.
\]
Consequently, if $A:=T_M[P_A]-T_P^{\mathrm{cap}}$ satisfies $\lambda_{\min}(A)\ge \delta>0$ and $\|T_P-T_P^{\mathrm{cap}}\|_F\le \varepsilon$, then
\[
  \lambda_{\min}\bigl(T_M[P_A]-T_P\bigr)\ \ge\ \delta-\varepsilon.
\]
\end{corollary}

\begin{lemma}[Szeg\H{o}--B\"ottcher barrier with explicit modulus]\label{lem:a3-sb-barrier}
Let $P_A$ be the Archimedean symbol constructed in Section~\ref{subsec:a3-symbol-floor}.
There exists an absolute constant $C_{\mathrm{SB}}=4$ such that for every $M\ge1$
\[
  \lambda_{\min}\!\bigl(T_M[P_A]\bigr)\ \ge\ \min_{\theta\in\TT}P_A(\theta)\;-\;C_{\mathrm{SB}}\,\omega_{P_A}\!\Bigl(\frac{1}{2M}\Bigr).
\]
\end{lemma}

\begin{remark}[Sources and scope of $C_{\mathrm{SB}}$]\label{rem:a3-sb-sources}
This is the classical Toeplitz eigenvalue stability for Lipschitz symbols.  We use the version recorded in
B\"ottcher--Silbermann's \textit{Introduction to Large Truncated Toeplitz Matrices} (Theorem~5.5 together with
Corollary~5.7 in Chapter~5); see also Grenander--Szeg\H{o} (Ch.~3) and Varga's \textit{Gershgorin and His Circles} (Cor.~2.5.3) for related Gershgorin-based formulations.
For the Lipschitz/H\"older classes relevant here the constant in front of the modulus is $C_{\mathrm{SB}}=4$.

Lemma~\ref{lem:a3-sb-barrier} is the only place where this numerical constant enters our treatment of A3.
Coupled with the uniform RKHS cap (Corollary~\ref{cor:uniform-prime-cap}) and the uniform floor $c_*>0$,
it yields the mixed lower bound in Theorem~\ref{thm:A3}.
\end{remark}

\begin{remark}[Operator difference vs.\ symbol difference]\label{rem:a3-operator-diff}
When applying Lemma~\ref{lem:a3-sb-barrier} we always work with
the Toeplitz operators $T_M[P_A]$ and $T_M[P_A]-T_P$; no ``symbol minus symbol'' simplification
is invoked.  The lower bounds track the operator difference directly, so all perturbative terms
are measured in operator/Frobenius norms as mandated by Lemma~\ref{lem:hw-kf}.
\end{remark}

\begin{remark}[Frobenius norm structure]
For Hermitian Toeplitz matrices with first row $c_0,\dots,c_{M-1}$ and coefficients $c_{-k}=\overline{c_k}$, one has
\(\|T_M[P]\|_F^2 = M|c_0|^2 + 2\sum_{k=1}^{M-1} (M-k)|c_k|^2\).
Hence a split budget $\varepsilon = \varepsilon^{\mathrm{F}}_{\text{tail}} + \varepsilon^{\mathrm{F}}_{\text{grid}} + \varepsilon^{\mathrm{F}}_{\text{num}}$ controls the total spectral drift of $T_P$ relative to the capped operator.
\end{remark}
