\subsection{Main closure: from analytic modules to Weil positivity}\label{sec:Main-closure}

\subsection*{Standing hypotheses (analytic chain)}
Throughout this section we rely only on the following proved ingredients:
\begin{itemize}
  \item \textbf{(T0) Normalization.} Guinand--Weil crosswalk and our conventions, cf.\ Proposition~\ref{prop:T0-GW} (Section~\ref{sec:T0}).
  \item \textbf{(A1$'$) Density.} The Fej\'er$\times$heat cone is dense in $W_K$, cf.\ Theorem~\ref{a1:thm:A1-local-density}.
  \item \textbf{(A2) Continuity.} The Weil functional $Q$ is continuous on $W_K$ with a modulus $L_Q(K)$ (Section~\ref{sec:A2}).
  \item \textbf{(A3) Toeplitz bridge.} For $M\ge M_0(K)$ one has
  \[
    \lambda_{\min}\!\big(T_M[P_A]-\mathcal T_P\big)\ \ge\ c_0(K)-C_T\,\omega_{P_A}\!\Big(\tfrac{\pi}{M}\Big)-\|\mathcal T_P\|,
  \]
  with analytic $c_0(K)\ge c_*\ge 811/1000$ (uniform floor from Lemma~\ref{lem:uniform-arch-floor}), $\omega_{P_A}$, $C_T$, cf.\ Theorem~\ref{thm:A3}.
  \item \textbf{(RKHS) Prime contraction.} For $t\ge t^\star_{\mathrm{rkhs}}(K)$ one has
  \(
    \|\mathcal T_P\|\le \rho\!\big(t^\star_{\mathrm{rkhs}}(K)\big)\le \tfrac14\,c_0(K),
  \)
  cf.\ Theorem~\ref{thm:rkhs-tstar} (Section~\ref{sec:rkhs-prime-cap}).
  \item \textbf{(T5) Compact transfer.} With the monotone schedules $t^\star_{\mathrm{T5}}(K)$, $M^\star(K)$ from \eqref{eq:T5-tstar}--\eqref{eq:T5-Mstar}, one has
  \(
    \lambda_{\min}\!\big(T_{M^\star(K)}[P_A]-\mathcal T_P\big)\ge\tfrac12\,c_0^\ast(K),
  \)
  hence $Q\ge0$ on $W_K$, cf.\ Theorem~\ref{thm:T5-compact}.
\end{itemize}

\begin{theorem}[Main positivity]\label{thm:Main-positivity}
If \textup{(T0)}+\textup{(A1$'$)}+\textup{(A2)}+\textup{(A3)}+\textup{(RKHS)}+\textup{(T5)} hold, then
\[
  Q(\Phi)\ \ge\ 0\qquad\text{for every even, real, compactly supported }\Phi\in \mathcal W,
\]
where $\mathcal W=\bigcup_{K>0}\mathcal W_K$ is the Weil cone from Section~\ref{sec:notation}.
\end{theorem}

\begin{proof}
Fix $K>0$. By \textup{(T5)} with the monotone schedules $t^\star_{\mathrm{T5}}(K)$, $M^\star(K)$, Lemma~\ref{lem:T5p-grid} together with Theorem~\ref{thm:A3} yield
\[
  \lambda_{\min}\!\big(T_{M^\star(K)}[P_A]-\mathcal T_P\big)\ \ge\ \tfrac12\,c_0^\ast(K)\ >\ 0.
\]
Hence the finite Toeplitz form is nonnegative on the Fej\'er$\times$heat cone. By \textup{(A1$'$)} the cone is dense in $W_K$, and by \textup{(A2)} the functional $Q$ is continuous; therefore $Q\ge0$ on $W_K$. Taking the union over all $K$ shows $Q\ge0$ on $W$. Finally \textup{(T0)} identifies this $Q$ with the canonical Weil functional.
\end{proof}

\begin{remark}[No numerics, no ATP]
The proof of Theorem~\ref{thm:Main-positivity} uses only analytic bounds established in Sections~\ref{sec:T0}--\ref{sec:T5}; legacy numerical certificates and ATP logs are archived separately for reproducibility but play no role in the argument.
\end{remark}
