% D3: From dispersion to operator contraction
\subsection{D3: Operator Bridge to \texorpdfstring{$\|T_P\|\le 1-\delta_0$}{||T_P|| ≤ 1−δ0}}\label{d3:sec:D3-operator}

The linear-algebraic bounds quoted here are standard consequences of Gershgorin and Rayleigh estimates~\cite{HornJohnson2013,Varga2004}.
\SeeAlso{D3 dispersion (Lemma~\ref{d3:lem:D3-dispersion}), mixed bound (Theorem~\ref{thm:A3}).}

Let $\mathcal H_K$ be the even RKHS on $[-K,K]$ with normalized kernels $\|k_\alpha\|=1$, and set
\(
T_P = \sum_{\alpha_n\in[-K,K]} w(n)\,|k_{\alpha_n}\rangle\!\langle k_{\alpha_n}|
\)
with $w(n)=\Lambda(n)/\sqrt n$.

\begin{lemma}[Dispersion via A2/A3 data]\label{d3:lem:D3-dispersion}
Assume the A3 hypotheses: $P_A\in\Lip(1)$ with $\min P_A\ge \czero>0$ (Lemmas~\ref{lem:a3.bv-to-lip}, \ref{lem:a3.lip-floor}), the trace-cap bound $\|T_P\|\le \rhok$ (Lemma~\ref{lem:trace-cap-bound}), and the two-scale construction of Lemma~\ref{lem:a3.two-scale}. Then there exist scales $t_{\mathrm{sym}}, t_{\mathrm{rkhs}}$ and a sequence $\delta_A\to0$ such that for every even RKHS test $f$ supported in $[-K,K]$
\[
  \Biggl|\sum_{p\le A}\Bigl(f(p)-\mathbb{E}_{\mathcal{P}\cap[1,A]}f\Bigr)\Biggr|
  \le C(K)\Bigl(\omega_{P_A}(t_{\mathrm{sym}})+\epsK(t_{\mathrm{rkhs}})\Bigr)
  \eqqcolon C(K)\,\delta_A.
\]
Consequently, $\delta_A\to0$ as $A\to\infty$.
\end{lemma}

\begin{proof}
The Lipschitz control from Lemmas~\ref{lem:a3.bv-to-lip} and \ref{lem:a3.sup-bounds} bounds the near-diagonal contribution by $\omega_{P_A}(t_{\mathrm{sym}})$.  The trace-cap bound (Lemma~\ref{lem:trace-cap-bound}) together with Lemma~\ref{lem:a3.two-scale} controls the RKHS tail by $\epsK(t_{\mathrm{rkhs}})$.  Adding the two estimates yields the desired inequality.
\end{proof}

\begin{theorem}[D3: Structural contraction]\label{d3:thm:D3-contraction}
If Lemma~\ref{d3:lem:D3-dispersion} provides a gain $\delta_\ast>0$ after fixing the scales, then there exists $\delta_0\in(0,\delta_\ast)$ with
\begin{equation}\label{eq:D3_operator_bridge-formula}
  \|T_P\|_{\mathcal H_K}\ \le\ 1-\delta_0.
\end{equation}
Inserting this into the mixed Toeplitz bound with Lipschitz symbol $P_A$ yields, for $M\gg K^3$,
\begin{equation}\label{eq:D3_operator_bridge-lambda-min}
  \lambda_{\min}(T_M[P_A]-T_P)\ \ge\ (1+\delta_0)\,\log(1{+}K)\ -\ O(1).
\end{equation}
\end{theorem}

\begin{proof}[Sketch]
In the packet basis the matrix of $T_P$ is $W^{1/2}GW^{1/2}$; the dispersion bound forces its Rayleigh quotients below $1-\delta_0$.  The remainder follows by the mixed Toeplitz estimate.
\end{proof}

\begin{corollary}[Amplitude closure]\label{d3:cor:amplitude-closure}
With the auxiliary suppressors (Roads B/C) and Theorem~\ref{d3:thm:D3-contraction} we obtain
\(
  \Gamma(K) \ge (1+\delta_0)\,\log(1+K) - O(1),
\)
closing the amplitude gate.
\end{corollary}
