\subsection{AD Normalization (Unitary FT + L2 Packets)}
\label{ad:sec:AD-norm-fix}
We fix the unitary Fourier transform
\begin{equation}\label{eq:AD_normalization_fix-formula-3}
 \widehat f(\gamma)=\frac{1}{\sqrt{2\pi}}\int_{\RR} f(u)\,e^{-i\gamma u}\,du,\qquad \|f\|_{L^2}=\|\widehat f\|_{L^2}.
\end{equation}
For the AD scale set $s(\tau)=1+|\tau|$, $\sigma(\tau)=\sqrt{t_0}\,s(\tau)$ with fixed $t_0>0$, and define the L$^2$-normalized Gaussian packet
\begin{equation}\label{eq:AD_normalization_fix-formula-2}
 \psi_\tau(u)=\exp\!\Big(-\frac{u^2}{2\sigma(\tau)^2}\Big)\,e^{i\tau u}\,/\,\|\exp(-u^2/2\sigma(\tau)^2)\|_{2},\qquad \|\psi_\tau\|_2=1.
\end{equation}
Then
\begin{equation}\label{eq:AD_normalization_fix-formula-1}
 \widehat\psi_\tau(\gamma)=\pi^{-1/4}\,\sigma(\tau)^{1/2}\,\exp\!\Big(-\tfrac{\sigma(\tau)^2}{2}(\gamma-\tau)^2\Big),\quad \int_{\RR}|\widehat\psi_\tau|^2=1.
\end{equation}
Consequently, the zero-side diagonal contributes $\tfrac{1}{2\pi}\log(1{+}|\tau|)$ up to an $O(1)$ edge constant, and the Zero\,$\to$\,Prime bridge A3 yields
\begin{equation}\label{eq:AD_normalization_fix-formula}
 \Gamma(K)\ \ge\ \kappa_{\mathrm{A3}}(t_0)\,\Big(\tfrac{1}{2\pi}-\Lambda_0(t_0,\kappa)\Big)\,\log(1{+}K)\ -\ \kappa_{\mathrm{A3}}(t_0)\,C_{\mathrm{edge}}(t_0),
\end{equation}
with $\Lambda_0(t_0,\kappa)=2\sum_{m\ge1}e^{-t_0\kappa^2 m^2/8}$.
